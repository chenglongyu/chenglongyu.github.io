\subsection*{Normal extensions and separable extensions}

\begin{exer}[Recognizing normal extensions]
Determine whether each of the following extensions is normal, and justify your answer.
\begin{enumerate}
  \item $\Q(\sqrt{2})/\Q$.
  \item $\Q(\sqrt[3]{2})/\Q$.
  \item $\Q(\sqrt[4]{2})/\Q(\sqrt{2})$.

  \item $\Q(\sqrt[3]{2},\omega)/\Q$, where $\omega=e^{2\pi i/3}$.
  \item $\Q(\sqrt[3]{2},\omega)/\Q(\omega)$, where $\omega=e^{2\pi i/3}$
  

\end{enumerate}
\end{exer}

\begin{exer}[Intersection and compositum of normal extensions]\label{exer:normal-cap-comp}
Let $L_1/K$ and $L_2/K$ be two normal extensions inside a common algebraic closure $\bar K$. (You can assume $L_1$ and $L_2$ are finite over $K$ if you like, but it is not necessary.)
\begin{enumerate}
  \item Show that $L_1\cap L_2$ is normal over $K$.
  \item Show that the subfield $L_1L_2$ generated by $L_1$ and $L_2$ is normal over $K$.
  \item Is the analogue true for separable extensions? For Galois extensions?
\end{enumerate}
\end{exer}


\begin{exer}
For each of the following extensions, compute $[L:K]$, $|\Hom_K(L,\bar K)|$, and $|\Aut_K(L)|$, and  determine which of the properties (normal, separable, Galois) hold.
\begin{enumerate}
    \item $\Q(\sqrt{2}, \sqrt{3}) / \Q$.
    \item $\Q(\sqrt[4]{2}) / \Q$.
    \item $\F_{p^n} / \F_p$. (Hint: The Frobenius endomorphism $x \mapsto x^p$.)
    
    \item $\Q(\zeta_7)/\Q$, where $\zeta_7=e^{2\pi i/7}$.
\end{enumerate}
\end{exer}


\subsection*{Galois correspondence: computation and structure}

\begin{exer}
Let $L/K$ be a finite Galois extension. Prove that there are only finitely many intermediate fields $M$ with $K\subseteq M \subseteq L$. If we only assume $L/K$ is finite separable, does the same conclusion hold?
\end{exer}


\begin{exer}[Galois correspondence for $\Q(\sqrt{2},\sqrt{3})/\Q$]
Let $L=\Q(\sqrt{2},\sqrt{3})$.
\begin{enumerate}
  \item Show that $\mathrm{Gal}(L/\Q)\cong \Z/2\Z\times \Z/2\Z$ by describing the four automorphisms explicitly.
  \item List all subgroups of $G$ and the corresponding intermediate fields.
  \item Verify that every intermediate field is normal over $\Q$ (and explain why this follows from the fact that $G$ is abelian).
\end{enumerate}
\end{exer}


\begin{exer}[Compositum and intersection]
Let $L/K$ be a finite Galois extension and let $M_1, M_2$ be intermediate fields. Let $M_1M_2$ be subfield generated by $M_1$ and $M_2$.Prove that:
\begin{enumerate}
  \item $\mathrm{Gal}(L/(M_1 M_2)) = \mathrm{Gal}(L/M_1) \cap \mathrm{Gal}(L/M_2)$.
  \item $\mathrm{Gal}(L/(M_1\cap M_2))$ is generated by $\mathrm{Gal}(L/M_1)\cup \mathrm{Gal}(L/M_2)$.
\end{enumerate}
Apply this to $L=\Q(\sqrt{2},\sqrt{3})$, $M_1=\Q(\sqrt{2})$, $M_2=\Q(\sqrt{3})$ and verify directly.
\end{exer}


\begin{exer}[Fundamental theorem: degree formulas]
Let $L/K$ be a finite Galois extension with Galois group $G$, and let $M$ be an intermediate field with corresponding subgroup $H=\mathrm{Gal}(L/M)$. Show that: $[L:M]=|H|$ and $[M:K]=[G:H]$.
\end{exer}

\end{exer}

\begin{exer}[Galois correspondence for $x^3-2$]
Let $L$ be the splitting field of $x^3-2$ over $\Q$.
\begin{enumerate}
  \item Show that $\mathrm{Gal}(L/\Q)\cong S_3$ by describing the action of each automorphism on the roots $\sqrt[3]{2}$, $\sqrt[3]{2}\omega$, $\sqrt[3]{2}\omega^2$.
  \item Draw the subgroup lattice of $S_3$ and the corresponding intermediate fields.
  \item Which intermediate fields are Galois over $\Q$? Verify using the normal subgroup criterion.
\end{enumerate}
\end{exer}
